data <- read.csv("data/census_data/edu_attainment.csv", skip = 1) %>%
  janitor::clean_names() %>%
  mutate(
    edu_pop_total = total_pop_18_24 + total_pop_25 + total_pop_65,
    edu_pop_me = me_pop_18_24 + me_total_pop_25 + me_pop_65, 
    edu_pop_under65_total = total_pop_18_24 + (total_pop_25 - total_pop_65),
    edu_pop_under65_me = me_pop_18_24 + (me_total_pop_25 - me_pop_65),
    
    total_hs.under_18_24 = total_less_hs12_18_24 + total_hs_18_24,
    total_hs.to.ba_18_24 = total_some_col_18_24,
    total_ba.over_18_24 = total_col_18_24,
    
    total_hs.under_25.over = total_pop_25 - total_over_hs_25,
    total_hs.to.ba_25.over = total_over_hs_25 - total_ba_over_25,
    total_ba.over_25.over = total_ba_over_25,
    
    total_hs.under_65.over = total_pop_65 - total_over_hs_65,
    total_hs.to.ba_65.over= total_over_hs_65 - total_ba_over_65,
    total_ba.over_65.over = total_ba_over_65,
    
    total_hs.under_under.65 = total_hs.under_18_24 + (total_hs.under_25.over - total_hs.under_65.over),
    total_hs.to.ba_under.65 = total_hs.to.ba_18_24 + (total_hs.to.ba_25.over - total_hs.to.ba_65.over),
    total_ba.over_under.65 = total_ba.over_18_24 + (total_ba.over_25.over - total_ba.over_65.over),
  ) %>%
  select(
    zcta_zip, edu_pop_total, edu_pop_under65_total, total_pop_65,
    total_hs.under_under.65, total_hs.to.ba_under.65, total_ba.over_under.65,
    total_hs.under_65.over, total_hs.to.ba_65.over, total_ba.over_65.over) -> dataset_clean
